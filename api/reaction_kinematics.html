274        if self.s <= 0.0:

277

279        pcm2 = (self.s - self.m1**2 - self.m2**2) ** 2 - 4.0 * self.m1**2 * self.m2**2

281            self.nogo = True
282            return
283

285
286        # CM rapidity
287        acmratio = (math.sqrt(self.m2**2 + self.pcm**2) + self.pcm) / self.m2
288        self.cmrap = math.log(acmratio)
289        self.thesinh = math.sinh(self.cmrap)
290        self.thecosh = math.cosh(self.cmrap)
291
292        # final-state CM momentum

296            return

302        self.e04 = math.sqrt(self.pcmp**2 + self.m4**2)
303
304        # lab-frame extrema

306        self.emin3 = self.e03 * self.thecosh - self.pcmp * self.thesinh - self.m3
307        self.emax4 = self.e04 * self.thecosh + self.pcmp * self.thesinh - self.m4
308        self.emin4 = self.e04 * self.thecosh - self.pcmp * self.thesinh - self.m4

365    def _kinematics_at_coscm(self, coscm):
366        # Kinematic computations require these to have been initialized by _compute()
367        # For example if reaction is below threshold these values will not be computed
368        if (
369            self.pcmp is None
370            or self.thecosh is None
371            or self.e03 is None
372            or self.thesinh is None
373            or self.e04 is None
374        ):
375            raise ValueError("Kinematic quantities not computed")
376        sincm = math.sqrt(max(0.0, 1.0 - coscm**2))
377
378        ppar3 = self.pcmp * self.thecosh * coscm + self.e03 * self.thesinh
379        pperp3 = self.pcmp * sincm
380        ptot3 = math.hypot(ppar3, pperp3)
381
382        ppar4 = -self.pcmp * self.thecosh * coscm + self.e04 * self.thesinh
383        pperp4 = self.pcmp * sincm
384        ptot4 = math.hypot(ppar4, pperp4)
385
386        theta3 = math.acos(ppar3 / ptot3) if ptot3 > 0 else 0.0
387        theta4 = math.acos(ppar4 / ptot4) if ptot4 > 0 else 0.0
388
389        e3 = self.e03 * self.thecosh + coscm * self.pcmp * self.thesinh - self.m3
390        e4 = self.e04 * self.thecosh - coscm * self.pcmp * self.thesinh - self.m4
391
392        v3 = ptot3 / (e3 + self.m3)
393        v4 = ptot4 / (e4 + self.m4)
394
395        return {
396            "coscm": coscm,
397            "theta_cm": math.acos(coscm),
398            "theta3": theta3,
399            "theta4": theta4,
400            "e3": e3,
401            "e4": e4,
402            "v3": v3,
403            "v4": v4,

405            "p4": ptot4,

This class calculates the lab-frame and center-of-mass kinematics for a reaction: projectile + target → ejectile + recoil
