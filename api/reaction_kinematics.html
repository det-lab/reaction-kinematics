166    def _compute(self, ek_mev: float) -> None:
167        self._nogo = False
168        self._pcmp = self._thesinh = self._thecosh = self._e03 = self._e04 = None
169
170        # Mandelstam s
171        s = (self._m1 + self._m2) ** 2 + 2.0 * self._m2 * ek_mev
172        if s <= 0.0:
173            self._nogo = True
174            return
175
176        # initial CM momentum
177        pcm2 = (s - self._m1**2 - self._m2**2) ** 2 - 4.0 * self._m1**2 * self._m2**2
178        if pcm2 < 0:
179            self._nogo = True
180            return
181        pcm = math.sqrt(pcm2 / (4.0 * s))
182
183        # CM rapidity → boost parameters
184        acmratio = (math.sqrt(self._m2**2 + pcm**2) + pcm) / self._m2
185        cmrap = math.log(acmratio)
186        self._thesinh = math.sinh(cmrap)
187        self._thecosh = math.cosh(cmrap)
188
189        # final-state CM momentum
190        pcmp2 = (s - self._m3**2 - self._m4**2) ** 2 - 4.0 * self._m3**2 * self._m4**2
191        if pcmp2 < 0:
192            self._nogo = True
193            return
194        self._pcmp = math.sqrt(pcmp2 / (4.0 * s))
195
196        # CM total energies of outgoing particles
197        self._e03 = math.sqrt(self._pcmp**2 + self._m3**2)
198        self._e04 = math.sqrt(self._pcmp**2 + self._m4**2)
199
200        # lab-frame energy extrema
201        self._emax3 = self._e03 * self._thecosh + self._pcmp * self._thesinh - self._m3
202        self._emin3 = self._e03 * self._thecosh - self._pcmp * self._thesinh - self._m3
203        self._emax4 = self._e04 * self._thecosh + self._pcmp * self._thesinh - self._m4
204        self._emin4 = self._e04 * self._thecosh - self._pcmp * self._thesinh - self._m4
205
206        # reset max-angle quantities to sentinel values
207        self._theta3max = None
208        self._theta4max = None
209        self._e3atmaxang = None
210        self._e4atmaxang = None
211        self._cmcos3max = None
212        self._cmcos4max = None
213
214        # max ejectile lab angle (only exists when pcmp < m3 * sinh(rapidity))
215        thetatest3: float | None = None
216        if self._m3 > 0.0:
217            thetatest3 = self._pcmp / (self._m3 * self._thesinh)
218            if thetatest3 < 1.0:
219                self._theta3max = math.asin(thetatest3)
220                patmax = (self._e03 * math.cos(self._theta3max) * self._thesinh) / (
221                    1.0 + thetatest3**2 * self._thesinh**2
222                )
223                eatmax = math.sqrt(patmax**2 + self._m3**2)
224                self._e3atmaxang = eatmax - self._m3
225                self._cmcos3max = (eatmax - self._e03 * self._thecosh) / (
226                    self._pcmp * self._thesinh
227                )
228
229        # elastic case: forward-angle symmetry forces theta3max = 90°
230        if (self._m1 + self._m2) == (self._m3 + self._m4) and thetatest3 is not None:
231            if abs(thetatest3 - 1.0) < 1e-3:
232                self._theta3max = math.pi / 2.0
233                self._cmcos3max = -1.0
234                self._e3atmaxang = (
235                    self._e03 * self._thecosh
236                    + self._cmcos3max * self._pcmp * self._thesinh
237                    - self._m3
238                )
239
240        # max recoil lab angle
241        thetatest4: float | None = None
242        if self._m4 > 0.0:
243            thetatest4 = self._pcmp / (self._m4 * self._thesinh)
244            if thetatest4 < 1.0:
245                self._theta4max = math.asin(thetatest4)
246                patmax = (self._e04 * math.cos(self._theta4max) * self._thesinh) / (
247                    1.0 + thetatest4**2 * self._thesinh**2
248                )
249                eatmax = math.sqrt(patmax**2 + self._m4**2)
250                self._e4atmaxang = eatmax - self._m4
251                self._cmcos4max = (eatmax - self._e04 * self._thecosh) / (
252                    self._pcmp * self._thesinh
253                )
254
255        # elastic case: theta4max = 90°
256        if (self._m1 + self._m2) == (self._m3 + self._m4) and thetatest4 is not None:
257            if abs(thetatest4 - 1.0) < 1e-3:
258                self._theta4max = math.pi / 2.0
259                self._cmcos4max = 1.0
260                self._e4atmaxang = (
261                    self._e04 * self._thecosh
262                    - self._cmcos4max * self._pcmp * self._thesinh
263                    - self._m4
264                )

266    def _kinematics_at_coscm(self, coscm: float) -> dict[str, float]:
267        if (
268            self._pcmp is None
269            or self._thecosh is None
270            or self._e03 is None
271            or self._thesinh is None
272            or self._e04 is None
273        ):
274            raise ValueError("Kinematic state not computed — call _bind first")
275
276        sincm = math.sqrt(max(0.0, 1.0 - coscm**2))
277
278        ppar3 = self._pcmp * self._thecosh * coscm + self._e03 * self._thesinh
279        pperp3 = self._pcmp * sincm
280        ptot3 = math.hypot(ppar3, pperp3)
281
282        ppar4 = -self._pcmp * self._thecosh * coscm + self._e04 * self._thesinh
283        pperp4 = self._pcmp * sincm
284        ptot4 = math.hypot(ppar4, pperp4)
285
286        e3 = self._e03 * self._thecosh + coscm * self._pcmp * self._thesinh - self._m3
287        e4 = self._e04 * self._thecosh - coscm * self._pcmp * self._thesinh - self._m4
288
289        # dOmega_lab/dOmega_cm for each ejectile, i.e. the factor that converts a
290        # lab-frame differential cross section to the cm-frame one:
291        # dsigma/dOmega_cm = dsigma/dOmega_lab * jacobianN_lab
292        #
293        # Both are d(cos_theta_lab)/d(coscm), coscm being cos_theta_cm of particle 3.
294        # ppar4's sign convention (see above) flips the sign of the thecosh term
295        # relative to particle 3's formula.
296        if ptot3 > 0:
297            jacobian3_lab = (self._pcmp**2 / ptot3**2) * (
298                coscm * ppar3 / ptot3 + sincm * pperp3 * self._thecosh / ptot3
299            )
300        else:
301            jacobian3_lab = math.nan
302        if ptot4 > 0:
303            jacobian4_lab = (self._pcmp**2 / ptot4**2) * (
304                coscm * ppar4 / ptot4 - sincm * pperp4 * self._thecosh / ptot4
305            )
306        else:
307            jacobian4_lab = math.nan
308
309        return {
310            "cos_theta_cm": coscm,
311            "theta_cm": math.acos(coscm),
312            "theta3_lab": math.acos(ppar3 / ptot3) if ptot3 > 0 else 0.0,
313            "theta4_lab": math.acos(ppar4 / ptot4) if ptot4 > 0 else 0.0,
314            "energy3_lab": e3,
315            "energy4_lab": e4,
316            "velocity3_lab": ptot3 / (e3 + self._m3),
317            "velocity4_lab": ptot4 / (e4 + self._m4),
318            "momentum3_lab": ptot3,
319            "momentum4_lab": ptot4,
320            "jacobian3_lab": jacobian3_lab,
321            "jacobian4_lab": jacobian4_lab,
322        }

324    def _build_table(self) -> None:
325        """Build the interpolation table over the full CM angle grid."""
326        keys = [
327            "cos_theta_cm",
328            "theta_cm",
329            "theta3_lab",
330            "theta4_lab",
331            "energy3_lab",
332            "energy4_lab",
333            "velocity3_lab",
334            "velocity4_lab",
335            "momentum3_lab",
336            "momentum4_lab",
337            "jacobian3_lab",
338            "jacobian4_lab",
339        ]
340        table: dict[str, list[float]] = {k: [] for k in keys}
341        for coscm in np.linspace(-1.0, 1.0, self.n_cm_grid_points):
342            row = self._kinematics_at_coscm(coscm)
343            for k in keys:
344                table[k].append(row[k])
345        self._table = table

347    def kinematics_table_at_beam_energy(
348        self,
349        beam_energy: float,
350        *,
351        angle_unit: AngleUnit = AngleUnit.deg,
352        energy_unit: EnergyUnit = EnergyUnit.MeV,
353    ) -> dict[str, npt.NDArray[np.float64]]:
354        """
355        Compute full kinematics over a CM angle grid.
356
357        Parameters
358        ----------
359        beam_energy : float
360            Beam kinetic energy.
361        angle_unit : AngleUnit, optional
362            Unit for angle outputs ``theta_cm``, ``theta3_lab``, ``theta4_lab``
363            (default degrees).
364        energy_unit : EnergyUnit, optional
365            Unit of ``beam_energy`` (default MeV).
366
367        Returns
368        -------
369        dict[str, np.ndarray]
370            Keys: ``"cos_theta_cm"``, ``"theta_cm"``, ``"theta3_lab"``, ``"theta4_lab"``,
371            ``"energy3_lab"``, ``"energy4_lab"``, ``"velocity3_lab"``, ``"velocity4_lab"``,
372            ``"momentum3_lab"``, ``"momentum4_lab"``, ``"jacobian3_lab"``,
373            ``"jacobian4_lab"``. Angle keys are in ``angle_unit`` (default degrees).
374            ``jacobianN_lab`` is ``dOmegaN_lab/dOmega_cm``, i.e. the factor that
375            converts a lab-frame differential cross section to the cm-frame one:
376            ``dsigma/dOmega_cm = dsigma/dOmegaN_lab * jacobianN_lab``.
377
378        Raises
379        ------
380        ValueError
381            If the reaction is kinematically forbidden at this energy.
382        """
383        angle_unit = AngleUnit.from_any(angle_unit)
384        ek_mev = _parse_energy(beam_energy, energy_unit)
385        self._bind(ek_mev)
386        if self._nogo:
387            raise ValueError(f"Reaction kinematically forbidden at beam_energy={ek_mev} MeV")
388        keys = [
389            "cos_theta_cm",
390            "theta_cm",
391            "theta3_lab",
392            "theta4_lab",
393            "energy3_lab",
394            "energy4_lab",
395            "velocity3_lab",
396            "velocity4_lab",
397            "momentum3_lab",
398            "momentum4_lab",
399            "jacobian3_lab",
400            "jacobian4_lab",
401        ]
402        rows = [
403            self._kinematics_at_coscm(coscm)
404            for coscm in np.linspace(-1.0, 1.0, self.n_cm_grid_points)
405        ]
406        result = {k: np.array([row[k] for row in rows]) for k in keys}
407        for k in keys:
408            if k.startswith("theta"):
409                result[k] = result[k] / angle_unit.value
410        return result

412    def kinematics_at_beam_energy_and_angle(
413        self,
414        beam_energy: float,
415        angle_name: str,
416        angle_value: float,
417        *,
418        angle_unit: AngleUnit = AngleUnit.deg,
419        energy_unit: EnergyUnit = EnergyUnit.MeV,
420        duplicate_tol: float = 1e-6,
421    ) -> dict[str, list[float]]:
422        """
423        Interpolate kinematic quantities at a fixed beam energy and kinematic variable value.
424
425        Always returns lists to handle multi-valued cases (e.g. two ejectile
426        energies at the same lab angle).
427
428        Parameters
429        ----------
430        beam_energy : float
431            Beam kinetic energy.
432        angle_name : str
433            Independent variable name, can be one of ``"theta3_lab"``, ``"theta4_lab"``,
434            ``"theta_cm"``, ``"cos_theta_cm"``.
435        angle_value : float
436            Value to evaluate at. For angle keys (``theta*``), interpreted in
437            ``angle_unit`` (default degrees). For ``"cos_theta_cm"``, treated
438            as a dimensionless cosine — ``angle_unit`` is ignored.
439        angle_unit : AngleUnit, optional
440            Unit of ``angle_value`` for angle keys (default degrees).
441        energy_unit : EnergyUnit, optional
442            Unit of ``beam_energy`` (default MeV).
443        duplicate_tol : float, optional
444            Tolerance for merging near-duplicate solutions (default 1e-6).
445
446        Returns
447        -------
448        dict of str -> list
449            Full dict of all kinematic variables, each a list of solutions
450            sorted descending by ``energy3_lab``. Angle keys (``theta*``) are
451            returned in ``angle_unit`` (default degrees).
452
453        Raises
454        ------
455        ValueError
456            If ``beam_energy`` or ``angle_value`` is not finite, if the reaction is
457            kinematically forbidden at this energy, or if ``angle_value`` is outside
458            the physical range.
459
460        Examples
461        --------
462        >>> rxn = Reaction("p", "3H", "n", "3He")
463        >>> rxn.kinematics_at_beam_energy_and_angle(1.2, "theta3_lab", 30)
464        {'theta3_lab': [...], 'energy3_lab': [...], ...}
465        """
466        angle_unit = AngleUnit.from_any(angle_unit)
467        if angle_name.startswith("theta"):
468            angle_value = angle_value * angle_unit.value
469
470        if not math.isfinite(angle_value):
471            raise ValueError(f"{angle_name}={angle_value} is not a finite number")
472
473        ek_mev = _parse_energy(beam_energy, energy_unit)
474        self._bind(ek_mev)
475        if self._nogo:
476            raise ValueError(f"Reaction kinematically forbidden at beam_energy={ek_mev} MeV")
477
478        if self._table is None:
479            self._build_table()
480        assert self._table is not None
481
482        keys = list(self._table.keys())
483        xs = self._table[angle_name]
484
485        solutions = []
486
487        exact_idx = np.where(np.isclose(xs, angle_value, atol=1e-12))[0]
488        if len(exact_idx) > 0:
489            for i in exact_idx:
490                solutions.append({k: self._table[k][i] for k in keys})
491        else:
492            found = False
493            for i in range(len(xs) - 1):
494                x0, x1 = xs[i], xs[i + 1]
495                if (x0 - angle_value) * (x1 - angle_value) <= 0 and x0 != x1:
496                    found = True
497                    t = (angle_value - x0) / (x1 - x0)
498                    solutions.append(
499                        {
500                            k: self._table[k][i] + t * (self._table[k][i + 1] - self._table[k][i])
501                            for k in keys
502                        }
503                    )
504            if not found:
505                raise ValueError(f"{angle_name}={angle_value} outside physical range")
506
507        unique: list = []
508        for sol in solutions:
509            if not any(abs(sol["energy3_lab"] - u["energy3_lab"]) < duplicate_tol for u in unique):
510                unique.append(sol)
511        unique.sort(key=lambda s: s["energy3_lab"], reverse=True)
512
513        output = {k: [s[k] for s in unique] for k in keys}
514        for k in keys:
515            if k.startswith("theta"):
516                output[k] = [v / angle_unit.value for v in output[k]]
517        return output

519    def kinematics_curve_at_angle(
520        self,
521        beam_energy_array: Iterable[float],
522        theta3_lab: float,
523        *,
524        angle_unit: AngleUnit = AngleUnit.deg,
525        energy_unit: EnergyUnit = EnergyUnit.MeV,
526    ) -> list[dict[str, npt.NDArray[np.float64]]]:
527        """
528        Compute ejectile kinematics at a fixed ejectile lab angle (``theta3_lab``)
529        over a range of beam energies.
530
531        Returns two branches (high- and low-energy) as a list of two dicts. Each
532        dict contains arrays indexed by beam energy, with ``NaN`` where that branch
533        does not exist. Branch 0 is always the higher-energy solution.
534
535        Parameters
536        ----------
537        beam_energy_array : array-like
538            Beam energies to sweep.
539        theta3_lab : float
540            Fixed ejectile lab angle (``theta3_lab`` in the output dict).
541        angle_unit : AngleUnit, optional
542            Unit of ``theta3_lab`` input and ``theta4_lab`` output (default degrees).
543        energy_unit : EnergyUnit, optional
544            Unit of ``beam_energy_array`` values (default MeV).
545
546        Returns
547        -------
548        list of two dicts, each with keys:
549            ``"beam_energy_lab"``, ``"energy3_lab"``, ``"energy4_lab"``, ``"theta4_lab"``,
550            ``"velocity3_lab"``, ``"velocity4_lab"``, ``"momentum3_lab"``, ``"momentum4_lab"``,
551            ``"jacobian3_lab"``, ``"jacobian4_lab"``.
552            ``theta4_lab`` is in ``angle_unit`` (default degrees).
553
554        Examples
555        --------
556        >>> rxn = Reaction("p", "3H", "n", "3He")
557        >>> branches = rxn.kinematics_curve_at_angle(np.linspace(1.0, 5.0, 200), 30)
558        >>> for b in branches:
559        ...     plt.plot(b["beam_energy_lab"], b["energy3_lab"])
560        """
561        angle_unit = AngleUnit.from_any(angle_unit)
562        theta_rad = theta3_lab * angle_unit.value
563
564        keys = [
565            "energy3_lab",
566            "energy4_lab",
567            "theta4_lab",
568            "velocity3_lab",
569            "velocity4_lab",
570            "momentum3_lab",
571            "momentum4_lab",
572            "jacobian3_lab",
573            "jacobian4_lab",
574        ]
575        branches = [
576            {"beam_energy_lab": [], **{k: [] for k in keys}},
577            {"beam_energy_lab": [], **{k: [] for k in keys}},
578        ]
579
580        for ek in beam_energy_array:
581            ek_mev = _parse_energy(ek, energy_unit)
582            try:
583                row = self.kinematics_at_beam_energy_and_angle(
584                    ek_mev, "theta3_lab", theta_rad, angle_unit=AngleUnit.rad
585                )
586            except ValueError:
587                solutions = []
588            else:
589                n = len(row["energy3_lab"])
590                solutions = [{k: row[k][i] for k in keys} for i in range(n)]
591
592            for i, branch in enumerate(branches):
593                branch["beam_energy_lab"].append(ek_mev)
594                sol = solutions[i] if i < len(solutions) else None
595                for k in keys:
596                    branch[k].append(sol[k] if sol is not None else float("nan"))
597
598        result = []
599        for branch in branches:
600            converted = {}
601            for k, v in branch.items():
602                arr = np.array(v)
603                if k.startswith("theta"):
604                    arr = arr / angle_unit.value
605                converted[k] = arr
606            result.append(converted)
607        return result

Defines a two-body nuclear reaction: projectile + target → ejectile + recoil.

All energy-dependent methods accept a beam_energy parameter (beam kinetic energy, MeV by default). Internal computation is cached per energy so repeated calls at the same energy are efficient.

Parameters
  • mass1 (str, MassInput, or float): Projectile mass, or a full reaction string in "target(beam,ejectile)recoil" notation (e.g. "3H(p,n)3He"). If notation is given, mass2–mass4 must be omitted.
  • mass2, mass3, mass4 (str, MassInput, or float, optional): Target, ejectile, and recoil masses. Required when mass1 is not a reaction notation string. Strings like "p", "12C", "alpha" are looked up in the mass table. Floats require mass_unit.
  • mass_unit (str or EnergyUnit, optional): Unit for numeric masses (e.g. "MeV", "keV").
Attributes
  • n_cm_grid_points (int): Total number of CM angle grid points (default 1001). Changing this invalidates the cache.
