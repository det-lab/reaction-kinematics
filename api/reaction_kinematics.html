130    def _compute(self, ek_mev: float) -> None:
131        self._nogo = False
132        self._pcmp = self._thesinh = self._thecosh = self._e03 = self._e04 = None
133
134        # Mandelstam s
135        s = (self._m1 + self._m2) ** 2 + 2.0 * self._m2 * ek_mev
136        if s <= 0.0:
137            self._nogo = True
138            return
139
140        # initial CM momentum
141        pcm2 = (s - self._m1**2 - self._m2**2) ** 2 - 4.0 * self._m1**2 * self._m2**2
142        if pcm2 < 0:
143            self._nogo = True
144            return
145        pcm = math.sqrt(pcm2 / (4.0 * s))
146
147        # CM rapidity → boost parameters
148        acmratio = (math.sqrt(self._m2**2 + pcm**2) + pcm) / self._m2
149        cmrap = math.log(acmratio)
150        self._thesinh = math.sinh(cmrap)
151        self._thecosh = math.cosh(cmrap)
152
153        # final-state CM momentum
154        pcmp2 = (s - self._m3**2 - self._m4**2) ** 2 - 4.0 * self._m3**2 * self._m4**2
155        if pcmp2 < 0:
156            self._nogo = True
157            return
158        self._pcmp = math.sqrt(pcmp2 / (4.0 * s))
159
160        # CM total energies of outgoing particles
161        self._e03 = math.sqrt(self._pcmp**2 + self._m3**2)
162        self._e04 = math.sqrt(self._pcmp**2 + self._m4**2)
163
164        # lab-frame energy extrema
165        self._emax3 = self._e03 * self._thecosh + self._pcmp * self._thesinh - self._m3
166        self._emin3 = self._e03 * self._thecosh - self._pcmp * self._thesinh - self._m3
167        self._emax4 = self._e04 * self._thecosh + self._pcmp * self._thesinh - self._m4
168        self._emin4 = self._e04 * self._thecosh - self._pcmp * self._thesinh - self._m4
169
170        # reset max-angle quantities to sentinel values
171        self._theta3max = None
172        self._theta4max = None
173        self._e3atmaxang = None
174        self._e4atmaxang = None
175        self._cmcos3max = None
176        self._cmcos4max = None
177
178        # max ejectile lab angle (only exists when pcmp < m3 * sinh(rapidity))
179        thetatest3: float | None = None
180        if self._m3 > 0.0:
181            thetatest3 = self._pcmp / (self._m3 * self._thesinh)
182            if thetatest3 < 1.0:
183                self._theta3max = math.asin(thetatest3)
184                patmax = (self._e03 * math.cos(self._theta3max) * self._thesinh) / (
185                    1.0 + thetatest3**2 * self._thesinh**2
186                )
187                eatmax = math.sqrt(patmax**2 + self._m3**2)
188                self._e3atmaxang = eatmax - self._m3
189                self._cmcos3max = (eatmax - self._e03 * self._thecosh) / (self._pcmp * self._thesinh)
190
191        # elastic case: forward-angle symmetry forces theta3max = 90°
192        if (self._m1 + self._m2) == (self._m3 + self._m4) and thetatest3 is not None:
193            if abs(thetatest3 - 1.0) < 1e-3:
194                self._theta3max = math.pi / 2.0
195                self._cmcos3max = -1.0
196                self._e3atmaxang = (
197                    self._e03 * self._thecosh
198                    + self._cmcos3max * self._pcmp * self._thesinh
199                    - self._m3
200                )
201
202        # max recoil lab angle
203        thetatest4: float | None = None
204        if self.m4 > 0.0:
205            thetatest4 = self._pcmp / (self._m4 * self._thesinh)
206            if thetatest4 < 1.0:
207                self._theta4max = math.asin(thetatest4)
208                patmax = (self._e04 * math.cos(self._theta4max) * self._thesinh) / (
209                    1.0 + thetatest4**2 * self._thesinh**2
210                )
211                eatmax = math.sqrt(patmax**2 + self.m4**2)
212                self._e4atmaxang = eatmax - self._m4
213                self._cmcos4max = (eatmax - self._e04 * self._thecosh) / (self._pcmp * self._thesinh)
214
215        # elastic case: theta4max = 90°
216        if (self._m1 + self._m2) == (self._m3 + self._m4) and thetatest4 is not None:
217            if abs(thetatest4 - 1.0) < 1e-3:
218                self._theta4max = math.pi / 2.0
219                self._cmcos4max = 1.0
220                self._e4atmaxang = (
221                    self._e04 * self._thecosh
222                    - self._cmcos4max * self._pcmp * self._thesinh
223                    - self._m4
224                )

226    def _kinematics_at_coscm(self, coscm: float) -> dict[str, float]:
227        if (
228            self._pcmp is None
229            or self._thecosh is None
230            or self._e03 is None
231            or self._thesinh is None
232            or self._e04 is None
233        ):
234            raise ValueError("Kinematic state not computed — call _bind first")
235
236        sincm = math.sqrt(max(0.0, 1.0 - coscm**2))
237
238        ppar3 = self._pcmp * self._thecosh * coscm + self._e03 * self._thesinh
239        pperp3 = self._pcmp * sincm
240        ptot3 = math.hypot(ppar3, pperp3)
241
242        ppar4 = -self._pcmp * self._thecosh * coscm + self._e04 * self._thesinh
243        pperp4 = self._pcmp * sincm
244        ptot4 = math.hypot(ppar4, pperp4)
245
246        e3 = self._e03 * self._thecosh + coscm * self._pcmp * self._thesinh - self._m3
247        e4 = self._e04 * self._thecosh - coscm * self._pcmp * self._thesinh - self._m4
248
249        return {
250            "coscm": coscm,
251            "theta_cm": math.acos(coscm),
252            "theta3": math.acos(ppar3 / ptot3) if ptot3 > 0 else 0.0,
253            "theta4": math.acos(ppar4 / ptot4) if ptot4 > 0 else 0.0,
254            "e3": e3,
255            "e4": e4,
256            "v3": ptot3 / (e3 + self._m3),
257            "v4": ptot4 / (e4 + self._m4),
258            "p3": ptot3,
259            "p4": ptot4,
260        }

262    def _build_table(self) -> None:
263        """Build the interpolation table over the full CM angle grid."""
264        keys = ["coscm", "theta_cm", "theta3", "theta4", "e3", "e4", "v3", "v4", "p3", "p4"]
265        table: dict[str, list[float]] = {k: [] for k in keys}
266        for i in range(-self._ncoscm, self._ncoscm + 1):
267            row = self._kinematics_at_coscm(i / self._ncoscm)
268            for k in keys:
269                table[k].append(row[k])
270        self._table = table

307    def at_value(
308        self,
309        x_name: str,
310        x: float,
311        *,
312        ek: float,
313        energy_unit: EnergyUnit = EnergyUnit.MeV,
314        y_names: str | list[str] | None = None,
315        duplicate_tol: float = 1e-6,
316    ) -> dict[str, list[float]]:
317        """
318        Interpolate kinematic quantities at a fixed independent variable.
319
320        Always returns lists to handle multi-valued cases (e.g. two ejectile
321        energies at the same lab angle).
322
323        Parameters
324        ----------
325        x_name : str
326            Independent variable name, e.g. ``"theta3"``, ``"theta_cm"``,
327            ``"coscm"``, ``"theta4"``.
328        x : float
329            Value to evaluate at (radians for angles).
330        ek : float
331            Beam kinetic energy.
332        energy_unit : EnergyUnit, optional
333            Unit of ``ek`` (default MeV).
334        y_names : str or list of str, optional
335            Dependent variables to return. ``None`` returns all.
336        duplicate_tol : float, optional
337            Tolerance for merging near-duplicate solutions (default 1e-6).
338
339        Returns
340        -------
341        dict of str -> list
342            Each value is a list of solutions, sorted descending by ``e3``.
343
344        Raises
345        ------
346        ValueError
347            If ``x`` is outside the physical range.
348
349        Examples
350        --------
351        >>> rxn = Reaction("p", "3H", "n", "3He")
352        >>> rxn.at_value("theta3", 0.5, ek=1.2, y_names=["e3", "v3"])
353        {'e3': [...], 'v3': [...]}
354        """
355        ek_mev = _parse_energy(ek, energy_unit)
356        self._bind(ek_mev)
357
358        if self._table is None:
359            self._build_table()
360        assert self._table is not None
361
362        xs = self._table[x_name]
363
364        if isinstance(y_names, str):
365            y_names = [y_names]
366        if y_names is None:
367            y_names = list(self._table.keys())
368
369        solutions = []
370
371        exact_idx = np.where(np.isclose(xs, x, atol=1e-12))[0]
372        if len(exact_idx) > 0:
373            for i in exact_idx:
374                solutions.append({k: self._table[k][i] for k in y_names})
375        else:
376            found = False
377            for i in range(len(xs) - 1):
378                x0, x1 = xs[i], xs[i + 1]
379                if (x0 - x) * (x1 - x) <= 0 and x0 != x1:
380                    found = True
381                    t = (x - x0) / (x1 - x0)
382                    solutions.append(
383                        {
384                            k: self._table[k][i] + t * (self._table[k][i + 1] - self._table[k][i])
385                            for k in y_names
386                        }
387                    )
388            if not found:
389                raise ValueError(f"{x_name}={x} outside physical range")
390
391        e_key = "e3" if "e3" in y_names else y_names[0]
392        unique: list = []
393        for sol in solutions:
394            if not any(abs(sol[e_key] - u[e_key]) < duplicate_tol for u in unique):
395                unique.append(sol)
396        unique.sort(key=lambda s: s[e_key], reverse=True)
397
398        return {k: [s[k] for s in unique] for k in y_names}

437        """
438        if isinstance(angle_unit, str):

354        """

356        self._bind(ek_mev)
357
358        if self._table is None:
359            self._build_table()
360        assert self._table is not None
361
362        xs = self._table[x_name]
363
364        if isinstance(y_names, str):
365            y_names = [y_names]
366        if y_names is None:
367            y_names = list(self._table.keys())
368
369        solutions = []
370
371        exact_idx = np.where(np.isclose(xs, x, atol=1e-12))[0]
372        if len(exact_idx) > 0:
373            for i in exact_idx:
374                solutions.append({k: self._table[k][i] for k in y_names})
375        else:
376            found = False
377            for i in range(len(xs) - 1):
378                x0, x1 = xs[i], xs[i + 1]
379                if (x0 - x) * (x1 - x) <= 0 and x0 != x1:
380                    found = True
381                    t = (x - x0) / (x1 - x0)
382                    solutions.append(
383                        {
384                            k: self._table[k][i] + t * (self._table[k][i + 1] - self._table[k][i])
385                            for k in y_names
386                        }
387                    )
388            if not found:
389                raise ValueError(f"{x_name}={x} outside physical range")
390
391        e_key = "e3" if "e3" in y_names else y_names[0]
392        unique: list = []
393        for sol in solutions:
394            if not any(abs(sol[e_key] - u[e_key]) < duplicate_tol for u in unique):
395                unique.append(sol)
396        unique.sort(key=lambda s: s[e_key], reverse=True)
397
398        return {k: [s[k] for s in unique] for k in y_names}
